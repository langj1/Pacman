6   package com.projekti.pacman.logiikka;
7   
8   import com.projekti.pacman.Suunta;
9   import java.awt.image.BufferedImage;
10  import java.io.IOException;
11  import java.io.InputStream;
12  import java.util.ArrayList;
13  import java.util.Random;
14  import javax.imageio.ImageIO;
15  
16  /**
17   * Luokka kuvaa Pacmania jahtaavia monstereita Perii luokan Liikkuva.
18   */
19  public class Monsteri extends Liikkuva {
20  
21      private int suuntaArvo;
22      private int vari;
23      private int valkkyy;
24  
25      /**
26       * SuuntaArvoa käytetään satunnaisen suunnan valinnassa.
27       *
28       * @param alkuX Aloitus x koordinaatti.
29       * @param alkuY Aloitus y koordinaatti.
30       * @param vari Kertoo monsterin värin.
31       */
32      public Monsteri(int alkuX, int alkuY, int vari) {
33          super(alkuX, alkuY);
34  
35          this.suuntaArvo = 0;
36          this.vari = vari;
37          this.valkkyy = 0;
38      }
39  
40      /**
41       * Metodi valitsee sattumanvaraisesti yhden suunnista, jotka sen parametrina
42       * saama lista sisältää. VASEN = 1, OIKEA = 2, YLOS = 3, ALAS = 4
43       *
44       * @param suunnat lista mahdollisista suunnista
45       */
46      public void arvoSuunta(ArrayList<Integer> suunnat) {
47  
48          Random random = new Random();
49  
50          while (true) {
51              int satunnaismuuttuja = random.nextInt(100);
52  
53              if (satunnaismuuttuja < 25) {
54  
55                  setSuunta(suunta.ALAS);
56                  this.suuntaArvo = 4;
57  
58              } else if (satunnaismuuttuja < 50) {
59  
60                  setSuunta(suunta.OIKEA);
61                  this.suuntaArvo = 2;
62  
63              } else if (satunnaismuuttuja < 75) {
64  
65                  setSuunta(suunta.VASEN);
66                  this.suuntaArvo = 1;
67  
68              } else {
69  
70                  setSuunta(suunta.YLOS);
71                  this.suuntaArvo = 3;
72              }
73  
74              if (suunnat.contains(this.suuntaArvo)) {
75                  break;
76              }
77          }
78      }
79  
80      @Override
81      public int getKenttaNumero() {
82          return 3;
83      }
84  
85      public int getSuuntaArvo() {
86          return suuntaArvo;
87      }
88  
89      public int getVari() {
90          return vari;
91      }
92  
93      public void setVari(int vari) {
94          this.vari = vari;
95      }
96  
97      @Override
98      public BufferedImage piirra() {
99  
100         BufferedImage ghostImg = null;
101         try {
102             if (vari == 1 && isSyotava()) {
103                 InputStream is = getClass().getClassLoader().getResourceAsStream("Images/Evading Ghost.png");
104                 ghostImg = ImageIO.read(is);
105             } else if (vari == 1) {
106                 InputStream is = getClass().getClassLoader().getResourceAsStream("Images/Orange Ghost.png");
107                 ghostImg = ImageIO.read(is);
108             }
109             if (vari == 2 && isSyotava()) {
110                 InputStream is = getClass().getClassLoader().getResourceAsStream("Images/Evading Ghost.png");
111                 ghostImg = ImageIO.read(is);
112             } else if (vari == 2) {
113                 InputStream is = getClass().getClassLoader().getResourceAsStream("Images/Teal Ghost.png");
114                 ghostImg = ImageIO.read(is);
115             }
116             if (vari == 3 && isSyotava()) {
117                 InputStream is = getClass().getClassLoader().getResourceAsStream("Images/Evading Ghost.png");
118                 ghostImg = ImageIO.read(is);
119             } else if (vari == 3) {
120                 InputStream is = getClass().getClassLoader().getResourceAsStream("Images/Pink Ghost.png");
121                 ghostImg = ImageIO.read(is);
122             }
123             if (vari == 4 && isSyotava()) {
124                 InputStream is = getClass().getClassLoader().getResourceAsStream("Images/Evading Ghost.png");
125                 ghostImg = ImageIO.read(is);
126             } else if (vari == 4) {
127                 InputStream is = getClass().getClassLoader().getResourceAsStream("Images/Red Ghost.png");
128                 ghostImg = ImageIO.read(is);
129             }
130         } catch (IOException e) {
131         }
132         return ghostImg;
133     }
134 
135 }
