4   
5   /**
6    * Luokka luo pohjan, jonka Pacman ja monsterit perii.
7    */
8   public abstract class Liikkuva {
9   
10      Suunta suunta;
11      private int xKordinaatti;
12      private int yKordinaatti;
13      private boolean pisteenPaalla;
14  
15      public Liikkuva(int alkuX, int alkuY) {
16  
17          this.suunta = suunta.STOP;
18          this.xKordinaatti = alkuX;
19          this.yKordinaatti = alkuY;
20          this.pisteenPaalla = false;
21  
22      }
23  
24      public void setPisteenPaalla(boolean pisteenPaalla) {
25          this.pisteenPaalla = pisteenPaalla;
26      }
27  
28      public boolean isPisteenPaalla() {
29          return pisteenPaalla;
30      }
31  
32      public Suunta getSuunta() {
33          return suunta;
34      }
35  
36      public void setSuunta(Suunta suunta) {
37          this.suunta = suunta;
38      }
39  
40      public int getxKordinaatti() {
41          return xKordinaatti;
42      }
43  
44      public int getyKordinaatti() {
45          return yKordinaatti;
46      }
47  
48      public int[] getKoordinaatit() {
49  
50          int[] koordinaatit = {yKordinaatti, xKordinaatti};
51  
52          return koordinaatit;
53      }
54  
55      public void setKoordinaatit(int x, int y) {
56          this.xKordinaatti = x;
57          this.yKordinaatti = y;
58      }
59  
60      /**
61       * Liikkuva liikkuu yhden askeleen tiettyyn suuntaan, jonka märittää suunta
62       * attribuutti.
63       */
64      public void liiku() {
65  
66          if (suunta == suunta.VASEN) {
67              this.xKordinaatti--;
68          }
69          if (suunta == suunta.OIKEA) {
70              this.xKordinaatti++;
71          }
72          if (suunta == suunta.YLOS) {
73              this.yKordinaatti--;
74          }
75          if (suunta == suunta.ALAS) {
76              this.yKordinaatti++;
77          }
78      }
79  
80      public int getKenttaNumero() {
81          return 4;
82      }
83  
84      public boolean onPacman() {
85          return false;
86      }
87  }
