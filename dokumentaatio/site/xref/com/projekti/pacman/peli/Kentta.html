6   package com.projekti.pacman.peli;
7   
8   import com.projekti.pacman.logiikka.Monsteri;
9   import com.projekti.pacman.logiikka.Pacman;
10  import java.util.ArrayList;
11  
12  /**
13   *
14   * @author langjimi
15   */
16  public abstract class Kentta {
17  
18      private int[][] kentta;
19      private int pisteet;
20  
21      public Kentta() {
22          kentta = new int[111][111];
23          pisteet = 0;
24      }
25  
26      public void setKentta(int[][] kentta) {
27          this.kentta = kentta;
28      }
29  
30      public int[][] getKentta() {
31          return kentta;
32      }
33  
34      public void asetaUusiArvo(int x, int y, int uusiArvo) {
35  
36          kentta[x][y] = uusiArvo;
37      }
38  
39      public int haePisteenArvo(int x, int y) {
40          return kentta[x][y];
41      }
42  
43      public int getPisteet() {
44  
45          return pisteet;
46      }
47  
48      public void setPisteet(int pisteet) {
49          this.pisteet = pisteet;
50      }
51  
52      public Pacman pacmaninLahtokohta() {
53          return new Pacman(1, 0);
54      }
55  
56      public ArrayList<Monsteri> monsterienLahtokohdat() {
57  
58          ArrayList<Monsteri> monsterit = new ArrayList<>();
59  
60          for (int i = 0; i < 4; i++) {
61              monsterit.add(new Monsteri(8 + i, 7));
62          }
63  
64          return monsterit;
65      }
66  }
